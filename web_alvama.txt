  function setupSectionWrapSlider() {
    const items = document.querySelectorAll(".section-wrap");
    let currentIndex = 0;

    // Detener cualquier intervalo previo para evitar conflictos
    if (window.sectionWrapInterval) {
      clearInterval(window.sectionWrapInterval);
    }

    // Si estamos en móvil
    if (items.length > 0 && window.innerWidth <= 767) {
      // Ocultar todos los items inicialmente
      items.forEach(item => {
        item.style.display = "none";
      });

      // Mostrar el primer item
      items[currentIndex].style.display = "block";

      // Crear función para cambiar al siguiente
      function showNextItem() {
        items[currentIndex].style.display = "none";
        currentIndex = (currentIndex + 1) % items.length;
        items[currentIndex].style.display = "block";
      }

      // Ejecutar el cambio cada 2 segundos
      window.sectionWrapInterval = setInterval(showNextItem, 2000);
    } else {
      // Restaurar comportamiento normal en pantallas más grandes
      items.forEach(item => {
        item.style.display = ""; // Deja que los estilos de Webflow se apliquen
      });
    }
  }

  // Ejecutar al cargar la página
